TAB DS 100*INTEGER
CZTERY DC INTEGER(4)
JEDEN DC INTEGER(1)
STO DC INTEGER(100)
ZERO DC INTEGER(0)
WYNIK DS INTEGER
INDEKS DS INTEGER
LA 1,TAB
L 2,0(1)
LR 3,2
LA 4,ZERO
L 5,0(4)
NAST A 5,JEDEN
C 5,STO
JZ KONIECT
A 1,CZTERY
L 2,0(1)
CR 3,2
JP ZAMIANA
JZ NAST
JN NAST
ZAMIANA LR 3,2
LR 6,5
J NAST
KONIECT ST 3,WYNIK
ST 6,INDEKS
KONIEC
